function fnPeriodYN(startdate, enddate) {

	try
	{
		if(enddate == '') {
			enddate = '999912310000';
		}

		var now = new Date();   //현재시간

		year = now.getFullYear();   //현재시간 중 4자리 연도
		month = now.getMonth()+1;   //현재시간 중 달. 달은 0부터 시작하기 때문에 +1

		if((month+"").length < 2){
			month="0"+month;   //달의 숫자가 1자리면 앞에 0을 붙임.
		}

		date = now.getDate();      //현재 시간 중 날짜.
		if((date+"").length < 2){
			date="0"+date;
		}

		hour = now.getHours();   //현재 시간 중 시간.
		if((hour+"").length < 2){
			hour="0"+hour;
		}

		minutes = now.getMinutes();   //현재 시간 중 분.
		if((minutes+"").length < 2){
			minutes="0"+minutes;
		}

		today = year + "" + month + "" + date+ "" +hour+ "" +minutes;      //오늘 날짜 완성.

		// 시간비교
		if (
				( eval(today) > eval(startdate) ) && ( eval(today) < eval(enddate) )
		)  {
			return "true";
		}
		else {

			return "false";
		}
	}
	catch (e)
	{
		return false;
	}
}
